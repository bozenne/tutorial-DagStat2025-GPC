\begin{align*}
	\widehat{U} - U = \underbrace{\frac{1}{m} \sum_{i=1}^{m} h_E(i)}_{\text{Experimental group}}&  + \underbrace{\frac{1}{n} \sum_{j=1}^{n} h_C(j)}_{\text{Control group}} + \underbrace{ \frac{1}{m n}  \sum_{i=1}^{n}\sum_{j=1}^{m} h_{EC}(i,j)}_{\text{Second order term}} \notag \\
	\text{where for } i \in \{1,\ldots,m\}, h_E(i)&=\mathbb{E}[\mathds{1}_{Y_i > X_j} - \mathds{1}_{X_i > Y_j}\mid X_i] - U \notag \\
	j \in \{1,\ldots,n\}, h_C(j)&=\mathbb{E}[\mathds{1}_{Y_i > X_j} - \mathds{1}_{X_i > Y_j}\mid Y_j] - U \notag  \\
 \widehat{\sigma}_{\widehat{U}} \underbrace{\approx}_{\text{First order}} \frac{1}{m^2} \sum_{i=1}^2 h^2_E(i) &+ \frac{1}{n^2} \sum_{j=1}^n h^2_C(j)
\end{align*}
